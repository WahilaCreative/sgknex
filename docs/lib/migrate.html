Validates some migrations by requiring and checking for an up and down function, returning an array with the item and migration require.

function validateMigrationStructure(migrator) { return function(item) { var migration = require(migrator.config.directory + '/' + item); if (!_.isFunction(migration.up) || !_.isFunction(migration.down)) { throw new Error('Invalid migration: ' + item + ' must have both an up and down function'); } return [item, migration]; }; }
